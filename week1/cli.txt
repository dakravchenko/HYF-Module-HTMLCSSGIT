(bonus) ls -R to see all nested folders 

touch a.txt 
echo "text" > a.txt (to write the first line)
echo "text2" >> a.txt (to write the second line, if just one ">" is used it will delete the first line)

touch a.txt b.txt 
echo "text" > a.txt
echo "text2" > b.txt
cat b.txt >> a.txt (to copy a content of one document to another, and again one > would delete the content of the first doc)
